    max_izquierda := _maximo(arreglo, 0, medio)
    max_derecha := _maximo(arreglo, medio + 1, len(arreglo) - 1)
    if max_izquierda > max_derecha {
        return max_izquierda
    } else {
        return max_derecha
    }

func _maximo(arreglo []int, inicio int, fin int) int {
    max := arreglo[inicio]

        if max < arreglo[i] {
            max = arreglo[i]
        }
    }
    return max
}
