func imprimirDyC(m int) {
    if m < 4 { (16 no es menor a 4)
        return
    }
    fmt.Println(m) (Printea 16)
    imprimirDyC(m / 4) (Manda M = 4 a la proxima iteracion)
    imprimirDyC(m - (m / 4)) (Manda M = 12 a la proxima iteracion)
}
